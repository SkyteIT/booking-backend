# Entity Framework Core Migration Commands
# Project: Ube Booking Backend
# Infrastructure Project: src\Ube.Infrastructure\Ube.Infrastructure.csproj
# Startup Project: src\Ube.Api\Ube.Api.csproj

# Check current migration status
dotnet ef migrations list --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Check if there are pending model changes
dotnet ef migrations has-pending-model-changes --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Create a new migration (replace MigrationName with descriptive name)
dotnet ef migrations add MigrationName --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Apply all pending migrations to database
dotnet ef database update --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Apply specific migration to database
dotnet ef database update MigrationName --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Generate SQL script for all migrations
dotnet ef migrations script --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Generate SQL script from specific migration to latest
dotnet ef migrations script MigrationName --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Generate SQL script between two specific migrations
dotnet ef migrations script FromMigration ToMigration --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Remove the last migration (if not applied to database)
dotnet ef migrations remove --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Check database connection
dotnet ef dbcontext info --project src\Ube.Infrastructure --startup-project src\Ube.Api

# List all DbContext types
dotnet ef dbcontext list --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Scaffold existing database (reverse engineering)
dotnet ef dbcontext scaffold "Server=localhost;Database=Ube_db;Trusted_Connection=True;" Microsoft.EntityFrameworkCore.SqlServer --project src\Ube.Infrastructure --startup-project src\Ube.Api --force

# Optimize the model snapshot
dotnet ef migrations optimize --project src\Ube.Infrastructure --startup-project src\Ube.Api

# Bundle migrations for deployment
dotnet ef migrations bundle --project src\Ube.Infrastructure --startup-project src\Ube.Api --output ./efbundle.exe</content>
